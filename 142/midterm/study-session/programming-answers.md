  ```java
      double balance = console.nextDouble();
  ```

  ```java
  public static void printGrid(int rows, int cols) {
      int n = 1;
      int count1 = 1;
      int count2 = 1;
      while (count1 <= rows * cols) {
          if (count1 % cols == 0) {
              System.out.println(n);
              count2++;
              n = count2;
          } else {
              System.out.print(n + ", ");
              n = n + rows;
          }
          count1++;
      }
  }
  ```

  ```java
  public static void threeHeads() {
      Random r = new Random();
      int numHeads = 0;
      while (numHeads < 3) {
          int flip = r.nextInt(2); // flip coin
          if (flip == 0) { // heads

          } else {

          }
  ```
