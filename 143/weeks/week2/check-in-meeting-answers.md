    public boolean isConsecutive(Stack<Integer> s) {
        if (s.size() < 2) {
            return false;
        }
        
        Queue<Integer> q = new LinkedList<Integer>();
        boolean isConsecutive = true;
        
        int curr = s.pop();
        while (!s.isEmpty()) {
            int next = s.pop();
            isConsecutive &= curr - 1 == next;
            q.add(curr);
            curr = next;
        }
        
        q.add(curr);
        
        while (!q.isEmpty()) {
            s.push(q.remove());
        }
        
        while (!s.isEmpty()) {
            q.add(s.pop());
        }
        
        while (!q.isEmpty()) {
            s.push(q.remove());
        }
        
        return isConsecutive;
    }
